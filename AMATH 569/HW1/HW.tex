\begin{problem}
    Solve the PDE:
    \begin{align*}
        \pp{u}{t} + u \pp{u}{x} = 0, ~~~ -\infty < x < \infty, t>0
    \end{align*}
    subject to the initial conditions:
    \begin{align*}
        u(x,0) = u_0(x) = \begin{cases}
            -1, &-\infty < x \leq -a, a > 0,\\
            \frac{x}{a}, &-a < x < a,\\
            1, &a \leq x < \infty.
        \end{cases}
    \end{align*}
\end{problem}

\begin{solution}
    
    \noindent
    Consider the homogeneous PDE
    \begin{align*}
        \pp{u}{t} + u \pp{u}{x} = 0, ~~~ -\infty < x < \infty, t>0
    \end{align*}
    subject to the initial conditions:
    \begin{align*}
        u(x,0) = u_0(x) = \begin{cases}
            -1, &-\infty < x \leq -a,\\
            \frac{x}{a}, &-a < x < a,\\
            1, &a \leq x < \infty.
        \end{cases}
    \end{align*}
    where $a > 0$. 



\end{solution}
